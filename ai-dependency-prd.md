## üìã Executive Summary

### Product Vision
Create an engaging, anonymous assessment tool that helps users understand their relationship with AI tools (ChatGPT, Claude, Gemini, Perplexity) through 10 adaptive questions that measure dependency levels while educating users about healthy AI usage patterns.

### Core Problem
With the explosion of LLM tools in 2024-2026, users have integrated ChatGPT, Claude, Gemini, and Perplexity into their daily lives without understanding:
- How dependent they've become on AI for basic thinking
- When AI use is helpful vs. harmful to cognitive development
- Their unique usage patterns across different AI tools
- Whether they're using AI strategically or as a cognitive crutch

### Solution
A 10-question interactive quiz that:
1. **Assesses** AI dependency across specific tools
2. **Educates** users about balanced AI usage
3. **Provides** personalized insights without judgment
4. **Collects** anonymous data to understand broader AI usage trends

## üìà Scoring Algorithm & Analysis

### Scoring Categories

```javascript
const scoringSystem = {
  categories: {
    chatgpt: {
      maxScore: 30,
      questions: [1, 2, 3, 5, 7, 9],
      weight: 1.5  // ChatGPT is most widely used
    },
    claude: {
      maxScore: 30,
      questions: [2, 3, 5, 9],
      weight: 1.3
    },
    gemini: {
      maxScore: 20,
      questions: [3, 8],
      weight: 1.0  // Lower adoption
    },
    general: {
      maxScore: 100,
      questions: [1, 4, 6, 7, 8, 10],
      weight: 1.2
    },
    awareness: {
      maxScore: 10,
      questions: [1, 2, 3, 5, 6, 10],
      weight: -1.0  // Inverse scoring (higher awareness = lower dependency)
    }
  }
};
```

### Overall Dependency Calculation

```javascript
function calculateOverallDependency(responses) {
  // Sum weighted scores from all categories
  const chatgptScore = (scores.chatgpt / 30) * 100;
  const claudeScore = (scores.claude / 30) * 100;
  const geminiScore = (scores.gemini / 20) * 100;
  const generalScore = (scores.general / 100) * 100;
  
  // Awareness reduces overall score
  const awarenessReduction = (scores.awareness / 10) * 20;
  
  // Calculate weighted average
  const rawScore = (
    (chatgptScore * 1.5) +
    (claudeScore * 1.3) +
    (geminiScore * 1.0) +
    (generalScore * 1.2)
  ) / 5.0;
  
  // Apply awareness reduction
  const finalScore = Math.max(0, rawScore - awarenessReduction);
  
  return Math.min(100, Math.round(finalScore));
}
```

## üìä Analytics & Data Collection

### Database Schema (Supabase PostgreSQL)

```sql
-- Anonymous Sessions Table
CREATE TABLE quiz_sessions (
  session_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  completed_at TIMESTAMP WITH TIME ZONE,
  device_type VARCHAR(20),
  browser VARCHAR(50),
  country_code VARCHAR(2),
  user_agent TEXT
);

-- Individual Responses Table
CREATE TABLE quiz_responses (
  response_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID REFERENCES quiz_sessions(session_id),
  question_id INTEGER NOT NULL,
  selected_option INTEGER NOT NULL,
  response_value INTEGER NOT NULL,
  category VARCHAR(20),
  time_spent_seconds INTEGER,
  timestamp TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Calculated Scores Table
CREATE TABLE quiz_scores (
  score_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID REFERENCES quiz_sessions(session_id),
  overall_score INTEGER NOT NULL,
  chatgpt_score INTEGER,
  claude_score INTEGER,
  gemini_score INTEGER,
  general_score INTEGER,
  awareness_score INTEGER,
  profile_tier VARCHAR(50),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Aggregated Analytics (Updated Daily via Cron)
CREATE TABLE daily_analytics (
  date DATE PRIMARY KEY,
  total_sessions INTEGER,
  completed_sessions INTEGER,
  completion_rate DECIMAL(5,2),
  avg_overall_score DECIMAL(5,2),
  avg_chatgpt_score DECIMAL(5,2),
  avg_claude_score DECIMAL(5,2),
  avg_gemini_score DECIMAL(5,2),
  tier_1_count INTEGER,  -- Independent Thinkers
  tier_2_count INTEGER,  -- Balanced Users
  tier_3_count INTEGER,  -- Friendly Dependent
  tier_4_count INTEGER,  -- AI Enthusiasts
  tier_5_count INTEGER,  -- Fully Integrated
  top_device_type VARCHAR(20),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Weekly Trends (For Longitudinal Analysis)
CREATE TABLE weekly_trends (
  week_start_date DATE PRIMARY KEY,
  avg_dependency_score DECIMAL(5,2),
  chatgpt_trend DECIMAL(5,2),  -- % change from previous week
  claude_trend DECIMAL(5,2),
  gemini_trend DECIMAL(5,2),
  total_completions INTEGER
);
```

### Key Analytics Queries

#### 1. **Overall Dependency Trend**
```sql
SELECT 
  date,
  avg_overall_score,
  avg_chatgpt_score,
  avg_claude_score,
  avg_gemini_score
FROM daily_analytics
WHERE date >= CURRENT_DATE - INTERVAL '30 days'
ORDER BY date;
```

#### 2. **Profile Distribution**
```sql
SELECT 
  profile_tier,
  COUNT(*) as user_count,
  ROUND(COUNT(*)::DECIMAL / SUM(COUNT(*)) OVER () * 100, 2) as percentage
FROM quiz_scores
WHERE created_at >= CURRENT_DATE - INTERVAL '7 days'
GROUP BY profile_tier
ORDER BY user_count DESC;
```

#### 3. **Most Problematic Questions** (Highest Dependency Indicators)
```sql
SELECT 
  question_id,
  AVG(response_value) as avg_score,
  COUNT(*) as times_asked,
  STDDEV(response_value) as score_variance
FROM quiz_responses
GROUP BY question_id
ORDER BY avg_score DESC;
```

#### 4. **ChatGPT vs Claude vs Gemini Usage Patterns**
```sql
SELECT 
  CASE 
    WHEN chatgpt_score > claude_score AND chatgpt_score > gemini_score THEN 'ChatGPT Dominant'
    WHEN claude_score > chatgpt_score AND claude_score > gemini_score THEN 'Claude Dominant'
    WHEN gemini_score > chatgpt_score AND gemini_score > claude_score THEN 'Gemini Dominant'
    ELSE 'Mixed Usage'
  END as primary_tool,
  COUNT(*) as user_count,
  AVG(overall_score) as avg_dependency
FROM quiz_scores
WHERE created_at >= CURRENT_DATE - INTERVAL '30 days'
GROUP BY primary_tool
ORDER BY user_count DESC;
```

#### 5. **Geographic Dependency Patterns**
```sql
SELECT 
  s.country_code,
  COUNT(*) as sessions,
  AVG(sc.overall_score) as avg_dependency,
  AVG(sc.chatgpt_score) as avg_chatgpt,
  AVG(sc.claude_score) as avg_claude
FROM quiz_sessions s
JOIN quiz_scores sc ON s.session_id = sc.session_id
WHERE s.completed_at IS NOT NULL
GROUP BY s.country_code
HAVING COUNT(*) > 100  -- Only countries with significant data
ORDER BY avg_dependency DESC;
```

### API Integration Example

```javascript
// supabase-client.js
import { createClient } from '@supabase/supabase-js'

const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL
const supabaseAnonKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY

export const supabase = createClient(supabaseUrl, supabaseAnonKey)

// Store quiz session
export async function storeQuizSession(sessionData) {
  // 1. Insert session
  const { data: session, error: sessionError } = await supabase
    .from('quiz_sessions')
    .insert({
      device_type: sessionData.deviceType,
      browser: sessionData.browser,
      country_code: sessionData.countryCode
    })
    .select()
    .single()

  if (sessionError) throw sessionError

  // 2. Insert responses
  const responsesToInsert = sessionData.responses.map(r => ({
    session_id: session.session_id,
    question_id: r.questionId,
    selected_option: r.selectedOption,
    response_value: r.value,
    category: r.category,
    time_spent_seconds: r.timeSpent
  }))

  const { error: responsesError } = await supabase
    .from('quiz_responses')
    .insert(responsesToInsert)

  if (responsesError) throw responsesError

  // 3. Calculate and insert scores
  const scores = calculateScores(sessionData.responses)
  
  const { error: scoresError } = await supabase
    .from('quiz_scores')
    .insert({
      session_id: session.session_id,
      overall_score: scores.overall,
      chatgpt_score: scores.chatgpt,
      claude_score: scores.claude,
      gemini_score: scores.gemini,
      general_score: scores.general,
      awareness_score: scores.awareness,
      profile_tier: scores.profileTier
    })

  if (scoresError) throw scoresError

  // 4. Mark session as completed
  await supabase
    .from('quiz_sessions')
    .update({ completed_at: new Date().toISOString() })
    .eq('session_id', session.session_id)

  return session.session_id
}
```

function createAnonymousSession() {
